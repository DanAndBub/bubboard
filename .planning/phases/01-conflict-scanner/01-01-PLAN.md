```typescript
function findMatches(content: string, keywords: string[]): string[] {
  const lower = content.toLowerCase();
  return keywords.filter(k => lower.includes(k.toLowerCase()));
}

function jaccard(a: Set<string>, b: Set<string>): number {
  const intersection = new Set([...a].filter(x => b.has(x)));
  const union = new Set([...a, ...b]);
  return union.size > 0 ? intersection.size / union.size : 0;
}
```

Three rules in `export const structuralRules: ConflictRule[]`:

1. `STRUCT_SUBAGENT_VISIBILITY` (severity: 'warning', category: 'structural'):
   - For each file NOT in SUBAGENT_VISIBLE set, scan content line-by-line
   - If a line matches any IMPERATIVE_PHRASES or AGENT_ACTION_KEYWORDS (via findMatches on that line), create a finding

   - Find the AGENTS.md file (match path case-insensitively)

     - Scan section content lines for IMPERATIVE_PHRASES via findMatches
     - If matches found, create a finding with `files: ['AGENTS.md']`, the section name, and matched phrases

3. `STRUCT_MODEL_ASSIGNMENT` (severity: 'warning', category: 'structural'):
   - Define model patterns: `['claude-opus', 'claude-sonnet', 'claude-haiku', 'gpt-4', 'gpt-3', 'gemini', 'deepseek']`
   - Define assignment indicators: `['|', '=', 'model:', 'use ', 'assigned', 'agent']`
   - For each file, scan lines. A line is a "model assignment" if it contains both a model pattern AND an assignment indicator
   - Collect `{file, model, line}` tuples
   - Group by approximate "role context" (the heading/section the line appears in)
   - If the same role context has different models across files, create a finding
